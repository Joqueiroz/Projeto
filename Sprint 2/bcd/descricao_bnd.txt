create database banco_salas;( indica criar um novo banco de dados, nome que você está dando ao novo banco de dados)
use banco_salas;(selecionar um banco de dados específico que você deseja usar para suas operações)

create table usuario(
id_usuario int auto_increment primary key,
name varchar(100) not null,(NOT NULL: coluna não pode conter valores nulos)
email varchar(255) not null unique,
cpf varchar(11) not null unique,
password varchar(50) not null
);

create table salas(
id_salas int auto_increment primary key,
descricao varchar(255) not null,
nome varchar(100) not null,
status_sala TINYINT not null, (para armazenar, Valores booleanos: Representando verdadeiro ou falso, códigos pequenos:
 Como status que não requerem um intervalo maior.
capacidade int( números inteiros) not null (tanto de pessoas q cabem na salas)
);

create table reserva(
id_reserva int auto_increment primary key,
data_hora datetime not null,
status_reserva TINYINT not null,
id_usuario int not null,
id_salas int not null,
foreign key (id_usuario) references usuario(id_usuario),
foreign key (id_salas) references salas(id_salas)
);
